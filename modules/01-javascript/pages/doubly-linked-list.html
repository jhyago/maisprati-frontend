02

Classe Node & inserirInicio

Inserir na frente — O(1)

O nó guarda o valor e dois ponteiros (ambos null ao nascer). Ao inserir no início: se a lista está vazia, o novo nó vira head e tail; senão, religamos os ponteiros entre o novo nó e o head atual.

lista_dupla.js 
class Node {
  constructor(valor) {
    this.valor    = valor;
    this.anterior = null;
    this.proximo  = null;
  }
}

class DoublyLinkedList {
  constructor() { this.head = null; this.tail = null; this.length = 0; }

  inserirInicio(valor) {
    const novoNo = new Node(valor);
    if (!this.head) { this.head = novoNo; this.tail = novoNo; }
    else {
      novoNo.proximo = this.head;
      this.head.anterior = novoNo;
      this.head = novoNo;
    }
    this.length++;
  }

  inserirFim(valor) {
    const novoNo = new Node(valor);
    if (!this.tail) { this.head = novoNo; this.tail = novoNo; }
    else {
      novoNo.anterior = this.tail;
      this.tail.proximo = novoNo;
      this.tail = novoNo;
    }
    this.length++;
  }

  imprimir() {
    let a = this.head, r = "null ⇄ ";
    while (a) { r += a.valor + " ⇄ "; a = a.proximo; }
    console.log(r + "null");
  }
}

const dll = new DoublyLinkedList();
dll.inserirFim(10); dll.inserirFim(20); dll.inserirInicio(5);
dll.imprimir(); // null ⇄ 5 ⇄ 10 ⇄ 20 ⇄ null

🔑 Por que religar os dois ponteiros? Numa lista simples bastaria novoNo.proximo = head. Na dupla, o antigo head também precisa saber quem é seu anterior — daí head.anterior = novoNo. Esquecer essa linha é o erro mais comum: a lista parece certa indo para frente, mas quebra ao voltar.
03

Método removerEm(indice)

Quatro casos a tratar

Remover por índice exige cuidado com os casos de borda. Validamos o índice e separamos em quatro situações — cada uma religa ponteiros diferentes.

removerEm.js 
removerEm(indice) {
  if (indice < 0 || indice >= this.length) {
    throw new RangeError("índice fora do intervalo");
  }

  if (this.length === 1) {       // único nó
    this.head = this.tail = null;
  } else if (indice === 0) {     // remove head
    this.head = this.head.proximo;
    this.head.anterior = null;
  } else if (indice === this.length - 1) { // remove tail
    this.tail = this.tail.anterior;
    this.tail.proximo = null;
  } else {                       // remove no meio
    let atual = this.head;
    for (let i = 0; i < indice; i++) atual = atual.proximo;
    atual.anterior.proximo = atual.proximo;
    atual.proximo.anterior = atual.anterior;
  }

  this.length--;
}

solução_A.js
visitar(url) {
  const nova = new Pagina(url);
  if (this.atual_no) {
    nova.anterior = this.atual_no;
    this.atual_no.proxima = nova; // descarta o futuro
  }
  this.atual_no = nova;
  return url;
}
voltar() {
  if (this.atual_no?.anterior) {
    this.atual_no = this.atual_no.anterior;
    return this.atual_no.url;
  }
  return null;
}
avancar() {
  if (this.atual_no?.proxima) {
    this.atual_no = this.atual_no.proxima;
    return this.atual_no.url;
  }
  return null;
}
